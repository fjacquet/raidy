    **ZFS ashift padding penalty (lines 536-544):**

    Add describe block: "Volumetry Engine - ZFS Ashift Padding Penalty"

    ZFS ashift parameter defines block alignment:
    - ashift=9: 512-byte sectors (2^9 = 512)
    - ashift=12: 4096-byte sectors (2^12 = 4096)
    - When ashift > physical sector size: padding waste

    Physical sector size mismatches:
    - 512B drives with ashift=12: 87.5% waste on small blocks (worst case)
    - Conservative 5% estimate per ashift level for typical workloads

    Test ZFS ashift padding:
    ```typescript
    it('should calculate ZFS ashift padding penalty when ashift > physical sector size', () => {
      const drive512B = { ...testDrives['hdd-10tb'], sector_size: 512 }

      const input: VolumetryInput = {
        topology: { type: 'zfs', level: 'raidz2' },
        driveCount: 12,
        drive: drive512B,
        serverCount: 1,
        zfsOptions: {
          ashift: 12, // 4096-byte alignment on 512-byte drives
          recordsize: 128 * 1024,
          compression: false,
          compressionRatio: 1.0,
          dedup: false,
          dedupRatio: 1.0,
        },
        // ... other options
      }
      const result = calculateVolumetry(input)

      // 12 drives × 10TB = 120TB raw
      // RAID-Z2: (12-2)/12 = 83.3% efficiency = 100TB
      // Ashift penalty: 12 - log2(512) = 12 - 9 = 3 levels above physical
      // Penalty: 100TB × 5% × 3 = 15TB overhead
      // After ashift penalty: 100TB - 15TB = 85TB
      // Slop space (1/32): 85TB / 32 ≈ 2.66TB (clamped to 128MiB - 128GiB)
      // Usable: 85TB - 2.66TB - filesystem overhead ≈ 81TB

      // Verify ashift penalty is applied (efficiency lower than expected)
      const baseEfficiency = (10 / 12) * 100 // 83.3% without ashift penalty
      expect(result.efficiency).toBeLessThan(baseEfficiency)

      // Verify ZFS details include ashift
      expect(result.zfsDetails?.ashift).toBe(12)
      expect(result.zfsDetails?.ashiftPaddingOverhead).toBeGreaterThan(0)
    })

    it('should not apply ZFS ashift penalty when ashift matches physical sector size', () => {
      const drive512B = { ...testDrives['hdd-10tb'], sector_size: 512 }

      const input: VolumetryInput = {
        topology: { type: 'zfs', level: 'raidz1' },
        driveCount: 6,
        drive: drive512B,
        serverCount: 1,
        zfsOptions: {
          ashift: 9, // Matches 512-byte physical sector (2^9 = 512)
          recordsize: 128 * 1024,
          compression: false,
          compressionRatio: 1.0,
          dedup: false,
          dedupRatio: 1.0,
        },
        // ... other options
      }
      const result = calculateVolumetry(input)

      // No ashift penalty when ashift matches physical sector size
      expect(result.zfsDetails?.ashiftPaddingOverhead).toBe(0)
    })

    it('should apply ZFS ashift penalty for 512B drives with ashift=13', () => {
      const drive512B = { ...testDrives['ssd-sata-2tb'], sector_size: 512 }

      const input: VolumetryInput = {
        topology: { type: 'zfs', level: 'mirror' },
        driveCount: 4,
        drive: drive512B,
        serverCount: 1,
        zfsOptions: {
          ashift: 13, // 8192-byte alignment on 512-byte drives
          recordsize: 128 * 1024,
          compression: false,
          compressionRatio: 1.0,
          dedup: false,
          dedupRatio: 1.0,
        },
        // ... other options
      }
      const result = calculateVolumetry(input)

      // ashift 13 - log2(512) = 13 - 9 = 4 levels above physical
      // Higher penalty than ashift=12 case
      expect(result.zfsDetails?.ashiftPaddingOverhead).toBeGreaterThan(0)
      expect(result.zfsDetails?.ashift).toBe(13)
    })
    ```
