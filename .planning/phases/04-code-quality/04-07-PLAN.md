```typescript
import { useTranslation } from "react-i18next";
import { Label, Slider, Select } from "@/components/common/FormControls";
import type { Topology } from "@/types";

interface DellOptionsPanelProps {
  topology: Topology & {
    type:
      | "powerflex"
      | "powerstore"
      | "powerscale"
      | "objectscale"
      | "powervault";
  };
  onChange: (topology: Topology) => void;
}

/**
 * Dell topology options panel.
 *
 * Consolidates configuration controls for all Dell storage topologies:
 * - PowerFlex: Failure groups, compression/dedup
 * - PowerStore: RAID levels, snapshots, inline reduction
 * - PowerScale: Node protection (N+x), snapshots
 * - ObjectScale: Erasure coding, geo-replication, overhead
 * - PowerVault: ADAPT RAID variants
 */
export function DellOptionsPanel({
  topology,
  onChange,
}: DellOptionsPanelProps) {
  const { t } = useTranslation("topology");

  // Helper to update topology-specific options
  const updateOptions = (key: string, value: number | string) => {
    const optionsKey = `${topology.type}Options`;
    onChange({
      ...topology,
      [optionsKey]: {
        ...(topology as any)[optionsKey],
        [key]: value,
      },
    });
  };

  // PowerFlex-specific controls
  if (topology.type === "powerflex") {
    const fgRatio = (topology as any).powerflexOptions?.fgRatio ?? 3;
    const compressionEnabled =
      (topology as any).powerflexOptions?.compressionEnabled ?? false;

    return (
      <div className="space-y-4">
        <div>
          <Label>{t("powerflex.fgRatio")}</Label>
          <Slider
            value={fgRatio}
            onChange={(value) => updateOptions("fgRatio", value)}
            min={2}
            max={6}
            step={1}
          />
          <span className="text-xs text-muted-foreground">
            {t("powerflex.fgRatio.help")}
          </span>
        </div>

        <div>
          <Label>{t("powerflex.compression")}</Label>
          <input
            type="checkbox"
            checked={compressionEnabled}
            onChange={(e) =>
              updateOptions("compressionEnabled", e.target.checked)
            }
          />
        </div>

        {/* Add dedup toggle */}
      </div>
    );
  }

  // PowerStore-specific controls
  if (topology.type === "powerstore") {
    const snapshotReservePercent =
      (topology as any).powerstoreOptions?.snapshotReservePercent ?? 15;

    return (
      <div className="space-y-4">
        <div>
          <Label>{t("powerstore.snapshotReserve")}</Label>
          <Slider
            value={snapshotReservePercent}
            onChange={(value) => updateOptions("snapshotReservePercent", value)}
            min={0}
            max={50}
            step={5}
          />
        </div>

        {/* Add dedup/compression toggles */}
      </div>
    );
  }

  // PowerScale-specific controls
  if (topology.type === "powerscale") {
    const nodeProtection =
      (topology as any).powerscaleOptions?.nodeProtection ?? "N+2";
    const snapshotReservePercent =
      (topology as any).powerscaleOptions?.snapshotReservePercent ?? 10;

    return (
      <div className="space-y-4">
        <div>
          <Label>{t("powerscale.nodeProtection")}</Label>
          <Select
            value={nodeProtection}
            onChange={(value) => updateOptions("nodeProtection", value)}
          >
            <option value="N+1">N+1 (single node failure)</option>
            <option value="N+2">N+2 (dual node failure)</option>
            <option value="N+3">N+3 (triple node failure)</option>
          </Select>
        </div>

        <div>
          <Label>{t("powerscale.snapshotReserve")}</Label>
          <Slider
            value={snapshotReservePercent}
            onChange={(value) => updateOptions("snapshotReservePercent", value)}
            min={0}
            max={30}
            step={5}
          />
        </div>
      </div>
    );
  }

  // ObjectScale-specific controls
  if (topology.type === "objectscale") {
    const ecK = (topology as any).objectscaleOptions?.ecK ?? 8;
    const ecM = (topology as any).objectscaleOptions?.ecM ?? 4;
    const geoReplicationFactor =
      (topology as any).objectscaleOptions?.geoReplicationFactor ?? 1;

    return (
      <div className="space-y-4">
        <div>
          <Label>{t("objectscale.ecConfiguration")}</Label>
          <div className="flex gap-2">
            <div>
              <Label className="text-xs">{t("objectscale.ecK")}</Label>
              <Select
                value={ecK}
                onChange={(value) => updateOptions("ecK", parseInt(value))}
              >
                <option value="4">4</option>
                <option value="8">8</option>
                <option value="12">12</option>
              </Select>
            </div>
            <div>
              <Label className="text-xs">{t("objectscale.ecM")}</Label>
              <Select
                value={ecM}
                onChange={(value) => updateOptions("ecM", parseInt(value))}
              >
                <option value="2">2</option>
                <option value="4">4</option>
              </Select>
            </div>
          </div>
        </div>

        <div>
          <Label>{t("objectscale.geoReplication")}</Label>
          <Select
            value={geoReplicationFactor}
            onChange={(value) =>
              updateOptions("geoReplicationFactor", parseInt(value))
            }
          >
            <option value="1">None (single site)</option>
            <option value="2">2-way geo-replication</option>
            <option value="3">3-way geo-replication</option>
          </Select>
        </div>
      </div>
    );
  }

  // PowerVault-specific controls (minimal)
  if (topology.type === "powervault") {
    return (
      <div className="space-y-4">
        <p className="text-sm text-muted-foreground">
          {t("powervault.adaptDescription")}
        </p>
      </div>
    );
  }

  return null;
}
```
