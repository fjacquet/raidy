/**
 * Tiering calculation module for hybrid storage configurations.
 *
 * Handles fast tier (SSD/NVMe) + capacity tier (HDD) configurations
 * for S2D, Nutanix, and Ceph hybrid deployments.
 */

import type { Drive } from '@/types'
import type { VolumetryStrategy } from '../strategies/VolumetryStrategy'

interface TieringConfig {
  fastTier: {
    drive: Drive
    driveCount: number
  }
  capacityTier: {
    drive: Drive
    driveCount: number
  }
}

interface TieringResult {
  fastTierRawCapacity: number
  fastTierUsableCapacity: number
  capacityTierRawCapacity: number
  capacityTierUsableCapacity: number
  totalUsableCapacity: number
  tieringBreakdown: Array<{
    name: string
    capacityBytes: number
    percentage: number
  }>
}

/**
 * Calculate capacity for hybrid tiered storage configurations.
 *
 * @param tieringConfig - Fast and capacity tier configuration
 * @param strategy - Topology strategy for capacity calculations
 * @param topologyLevel - Topology level (e.g., 'Mirror', 'ec_4_2')
 * @returns Tiering calculation results with breakdown
 *
 * @example
 * // S2D hybrid: 4x NVMe (fast) + 12x HDD (capacity)
 * const result = calculateTieredCapacity(
 *   {
 *     fastTier: { drive: nvmeDrive, driveCount: 4 },
 *     capacityTier: { drive: hddDrive, driveCount: 12 },
 *   },
 *   s2dStrategy,
 *   'Mirror'
 * )
 */
export function calculateTieredCapacity(
  tieringConfig: TieringConfig,
  strategy: VolumetryStrategy,
  topologyLevel: string,
  topologyOptions?: any
): TieringResult {
  const { fastTier, capacityTier } = tieringConfig

  // Calculate fast tier capacity
  const fastTierRaw = fastTier.drive.capacity_raw * fastTier.driveCount
  const fastTierDataFraction = strategy.calculateDataFraction(
    topologyLevel,
    fastTier.driveCount,
    topologyOptions
  )
  const fastTierOverhead = strategy.calculateOverhead?.(fastTierRaw, topologyOptions) ?? 0
  const fastTierUsable = fastTierRaw * fastTierDataFraction - fastTierOverhead

  // Calculate capacity tier capacity
  const capacityTierRaw = capacityTier.drive.capacity_raw * capacityTier.driveCount
  const capacityTierDataFraction = strategy.calculateDataFraction(
    topologyLevel,
    capacityTier.driveCount,
    topologyOptions
  )
  const capacityTierOverhead =
    strategy.calculateOverhead?.(capacityTierRaw, topologyOptions) ?? 0
  const capacityTierUsable = capacityTierRaw * capacityTierDataFraction - capacityTierOverhead

  // Total usable capacity
  const totalUsableCapacity = fastTierUsable + capacityTierUsable

  // Build breakdown
  const tieringBreakdown = [
    {
      name: 'Fast Tier (SSD/NVMe)',
      capacityBytes: fastTierUsable,
      percentage: (fastTierUsable / totalUsableCapacity) * 100,
    },
    {
      name: 'Capacity Tier (HDD)',
      capacityBytes: capacityTierUsable,
      percentage: (capacityTierUsable / totalUsableCapacity) * 100,
    },
  ]

  return {
    fastTierRawCapacity: fastTierRaw,
    fastTierUsableCapacity: fastTierUsable,
    capacityTierRawCapacity: capacityTierRaw,
    capacityTierUsableCapacity: capacityTierUsable,
    totalUsableCapacity,
    tieringBreakdown,
  }
}
```

/**
 * ObjectScale geo-replication overhead lookup tables.
 *
 * ObjectScale has three overhead layers:
 * 1. EC efficiency (from erasure coding k+m)
 * 2. Geo-replication overhead (multi-site redundancy)
 * 3. System overhead (~2-3% for metadata)
 *
 * These tables are validated against Dell ObjectScale documentation.
 */

interface GeoOverheadConfig {
  ecK: number
  ecM: number
  geoReplicationFactor: 1 | 2 | 3
}

interface GeoOverheadResult {
  ecEfficiency: number
  geoOverhead: number
  systemOverhead: number
  combinedEfficiency: number
}

/**
 * Geo-replication overhead lookup table.
 *
 * Maps replication factor to overhead multiplier.
 * - 1x (single site): 1.0 (no geo overhead)
 * - 2x (dual site): 2.0 (full duplication)
 * - 3x (triple site): 3.0 (triple redundancy)
 */
const GEO_REPLICATION_OVERHEAD: Record<1 | 2 | 3, number> = {
  1: 1.0, // No geo-replication
  2: 2.0, // 2-way geo-replication (full duplication)
  3: 3.0, // 3-way geo-replication (triple redundancy)
}

/**
 * System overhead for ObjectScale metadata.
 *
 * Approximately 2-3% depending on object size and configuration.
 * Conservative estimate: 2.5%
 */
const OBJECTSCALE_SYSTEM_OVERHEAD = 0.025

/**
 * Get ObjectScale geo-replication overhead and efficiency.
 *
 * @param config - EC and geo-replication configuration
 * @returns Overhead breakdown and combined efficiency
 *
 * @example
 * // 8+4 EC with 2-way geo-replication
 * const overhead = getObjectScaleGeoOverhead({
 *   ecK: 8,
 *   ecM: 4,
 *   geoReplicationFactor: 2,
 * })
 * // Returns:
 * // {
 * //   ecEfficiency: 0.667 (8/(8+4)),
 * //   geoOverhead: 2.0,
 * //   systemOverhead: 0.025,
 * //   combinedEfficiency: 0.325 (0.667 / 2.0 * (1 - 0.025))
 * // }
 */
export function getObjectScaleGeoOverhead(config: GeoOverheadConfig): GeoOverheadResult {
  const { ecK, ecM, geoReplicationFactor } = config

  // EC efficiency: k / (k + m)
  const ecEfficiency = ecK / (ecK + ecM)

  // Geo-replication overhead multiplier
  const geoOverhead = GEO_REPLICATION_OVERHEAD[geoReplicationFactor]

  // System overhead (metadata)
  const systemOverhead = OBJECTSCALE_SYSTEM_OVERHEAD

  // Combined efficiency accounting for all three factors
  // Formula: (EC efficiency / geo overhead) * (1 - system overhead)
  const combinedEfficiency = (ecEfficiency / geoOverhead) * (1 - systemOverhead)

  return {
    ecEfficiency,
    geoOverhead,
    systemOverhead,
    combinedEfficiency,
  }
}

/**
 * Validate ObjectScale EC configuration.
 *
 * Valid combinations:
 * - 4+2 (66.7% efficient)
 * - 8+4 (66.7% efficient)
 * - 12+4 (75% efficient)
 */
export function isValidObjectScaleEC(k: number, m: number): boolean {
  const validConfigurations = [
    [4, 2],
    [8, 4],
    [12, 4],
  ]

  return validConfigurations.some(([validK, validM]) => k === validK && m === validM)
}

/**
 * Filesystem overhead mapping for capacity calculations.
 *
 * Each filesystem has metadata overhead that reduces usable capacity
 * beyond the RAID/topology efficiency.
 */

export type FilesystemType = 'xfs' | 'ext4' | 'ntfs' | 'refs' | 'zfs' | 'none'

interface FilesystemOverhead {
  percentage: number
  description: string
}

/**
 * Filesystem overhead lookup table.
 *
 * Overhead percentages based on industry documentation:
 * - XFS: ~1.5% (inode tables, AG headers)
 * - ext4: ~1.2% (inode tables, group descriptors)
 * - NTFS: ~2% (MFT, system files)
 * - ReFS: ~1% (metadata checksum tables)
 * - ZFS: Handled by ZFS strategy (slop space)
 * - none: 0% (raw block device)
 */
const FILESYSTEM_OVERHEAD: Record<FilesystemType, FilesystemOverhead> = {
  xfs: {
    percentage: 0.015, // 1.5%
    description: 'XFS metadata (inode tables, AG headers)',
  },
  ext4: {
    percentage: 0.012, // 1.2%
    description: 'ext4 metadata (inode tables, group descriptors)',
  },
  ntfs: {
    percentage: 0.02, // 2%
    description: 'NTFS metadata (MFT, system files)',
  },
  refs: {
    percentage: 0.01, // 1%
    description: 'ReFS metadata (checksum tables)',
  },
  zfs: {
    percentage: 0, // Handled by ZFS strategy slop calculation
    description: 'ZFS overhead handled by slop space calculation',
  },
  none: {
    percentage: 0, // No filesystem overhead
    description: 'Raw block device (no filesystem)',
  },
}

/**
 * Get filesystem overhead for capacity calculation.
 *
 * @param filesystemType - Filesystem type
 * @returns Overhead multiplier (1 - percentage)
 *
 * @example
 * // XFS with 10TB usable capacity
 * const overhead = getFilesystemOverhead('xfs')
 * // overhead = 0.985 (98.5% efficiency)
 * const actualUsable = 10 * TB * overhead.multiplier
 * // ~9.85 TB actual usable after filesystem metadata
 */
export function getFilesystemOverhead(filesystemType: FilesystemType): {
  percentage: number
  multiplier: number
  description: string
} {
  const overhead = FILESYSTEM_OVERHEAD[filesystemType]

  return {
    percentage: overhead.percentage,
    multiplier: 1 - overhead.percentage,
    description: overhead.description,
  }
}
```
