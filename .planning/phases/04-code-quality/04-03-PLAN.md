```typescript
/**
 * Shared context for topology option panels.
 *
 * Provides topology state and onChange handler to child panels via React Context,
 * avoiding prop drilling through deeply nested form controls.
 */
import { createContext, useContext } from "react";
import type { ReactNode } from "react";
import type { Topology } from "@/types";

interface TopologyContextValue {
  topology: Topology;
  onChange: (topology: Topology) => void;
}

const TopologyContext = createContext<TopologyContextValue | null>(null);

/**
 * Custom hook to access topology context.
 * Must be used within TopologyProvider.
 */
export function useTopologyContext() {
  const context = useContext(TopologyContext);
  if (!context) {
    throw new Error("useTopologyContext must be used within TopologyProvider");
  }
  return context;
}

/**
 * Context provider for topology option panels.
 *
 * Wraps per-topology option panels to provide shared state access.
 */
export function TopologyProvider({
  topology,
  onChange,
  children,
}: {
  topology: Topology;
  onChange: (topology: Topology) => void;
  children: ReactNode;
}) {
  return (
    <TopologyContext.Provider value={{ topology, onChange }}>
      {children}
    </TopologyContext.Provider>
  );
}
```

**Why this approach:**

- Compound component pattern from React official docs and research
- Avoids prop drilling through nested form controls
- Each panel can access topology state via useTopologyContext() hook
- Type-safe: TypeScript enforces topology and onChange are provided
- Follows project conventions: named exports, TypeScript types, JSDoc

**Pattern reference:** Research section "Compound Component with Context" (React patterns.dev).

```typescript
import { useTranslation } from "react-i18next";
import { Label, Slider, Select } from "@/components/common/FormControls";
import { useTopologyContext } from "./shared/TopologyContext";
import type { Topology } from "@/types";

interface ZfsOptionsPanelProps {
  topology: Topology & { type: "zfs" };
  onChange: (topology: Topology) => void;
}

/**
 * ZFS-specific topology options panel.
 *
 * Displays configuration controls for ZFS topologies:
 * - ashift (sector size power)
 * - Compression ratio
 * - Recordsize
 * - ARC cache settings
 */
export function ZfsOptionsPanel({ topology, onChange }: ZfsOptionsPanelProps) {
  const { t } = useTranslation("topology");

  // Extract current ZFS options with defaults
  const ashift = topology.zfsOptions?.ashift ?? 12;
  const compressionRatio = topology.zfsOptions?.compressionRatio ?? 1.5;
  const recordsize = topology.zfsOptions?.recordsize ?? 128;

  // Helper to update specific ZFS option
  const updateZfsOption = (key: string, value: number) => {
    onChange({
      ...topology,
      zfsOptions: {
        ...topology.zfsOptions,
        [key]: value,
      },
    });
  };

  return (
    <div className="space-y-4">
      <div>
        <Label>{t("zfs.ashift")}</Label>
        <Slider
          value={ashift}
          onChange={(value) => updateZfsOption("ashift", value)}
          min={9}
          max={16}
          step={1}
        />
        <span className="text-xs text-muted-foreground">
          {t("zfs.ashift.help")}
        </span>
      </div>

      <div>
        <Label>{t("zfs.compressionRatio")}</Label>
        <Slider
          value={compressionRatio}
          onChange={(value) => updateZfsOption("compressionRatio", value)}
          min={1.0}
          max={3.0}
          step={0.1}
        />
      </div>

      {/* Add remaining ZFS options: recordsize, arc settings, etc. */}
    </div>
  );
}
```

```typescript
/**
 * Topology configuration panel - routing component.
 *
 * Displays topology type selector and conditionally renders
 * per-topology option panels based on selected type.
 *
 * Component reduced from 1647 lines to ~200-300 lines by extracting
 * topology-specific UI to dedicated panels.
 */
export function TopologyPanel() {
  const { topology, setTopology } = useConfigStore();
  const { t } = useTranslation("topology");

  return (
    <div className="space-y-5">
      {/* Topology type selector */}
      <div>
        <Label>{t("type.label")}</Label>
        <Select
          value={topology.type}
          onChange={(type) => setTopology({ ...topology, type })}
        >
          {/* All topology types */}
        </Select>
      </div>

      {/* Topology level selector */}
      <div>
        <Label>{t("level.label")}</Label>
        <Select
          value={topology.level}
          onChange={(level) => setTopology({ ...topology, level })}
        >
          {/* Levels for selected type */}
        </Select>
      </div>

      {/* Per-topology option panels */}
      {topology.type === "zfs" && (
        <ZfsOptionsPanel topology={topology} onChange={setTopology} />
      )}
      {topology.type === "vsan_esa" && (
        <VsanOptionsPanel topology={topology} onChange={setTopology} />
      )}
      {topology.type === "s2d" && (
        <S2dOptionsPanel topology={topology} onChange={setTopology} />
      )}
      {topology.type === "ceph" && (
        <CephOptionsPanel topology={topology} onChange={setTopology} />
      )}
      {topology.type === "nutanix" && (
        <NutanixOptionsPanel topology={topology} onChange={setTopology} />
      )}

      {/* Standard RAID and simpler topologies: keep inline or extract if needed */}
    </div>
  );
}
```
