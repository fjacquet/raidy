```typescript
/**
 * Bottleneck identification for performance calculations.
 *
 * Identifies the limiting factor in storage performance by comparing:
 * 1. Media limit (drive IOPS/bandwidth)
 * 2. Controller limit (HBA/RAID controller throughput)
 * 3. Bus limit (SAS/SATA/NVMe interface bandwidth)
 * 4. Network limit (storage network throughput)
 *
 * Returns the bottleneck name and effective throughput.
 */

export type BottleneckType =
  | "media"
  | "controller"
  | "bus"
  | "network"
  | "none";

interface BottleneckLimits {
  mediaIOPS: number;
  controllerIOPS?: number;
  busIOPS?: number;
  networkIOPS?: number;
}

interface BottleneckResult {
  bottleneck: BottleneckType;
  effectiveIOPS: number;
  limitingFactor: string;
}

/**
 * Identify performance bottleneck from multiple limit points.
 *
 * @param limits - IOPS limits for each layer (media, controller, bus, network)
 * @returns Bottleneck identification and effective IOPS
 *
 * @example
 * // Media: 12,000 IOPS, Controller: 8,000 IOPS, Bus: 10,000 IOPS, Network: 15,000 IOPS
 * const bottleneck = identifyBottleneck({
 *   mediaIOPS: 12000,
 *   controllerIOPS: 8000,
 *   busIOPS: 10000,
 *   networkIOPS: 15000,
 * })
 * // Returns: { bottleneck: 'controller', effectiveIOPS: 8000, limitingFactor: 'HBA/Controller' }
 */
export function identifyBottleneck(limits: BottleneckLimits): BottleneckResult {
  const { mediaIOPS, controllerIOPS, busIOPS, networkIOPS } = limits;

  // Build array of limits with their types
  const limitPoints: Array<{
    type: BottleneckType;
    iops: number;
    factor: string;
  }> = [{ type: "media", iops: mediaIOPS, factor: "Drive Media" }];

  if (controllerIOPS !== undefined) {
    limitPoints.push({
      type: "controller",
      iops: controllerIOPS,
      factor: "HBA/Controller",
    });
  }

  if (busIOPS !== undefined) {
    limitPoints.push({
      type: "bus",
      iops: busIOPS,
      factor: "Storage Bus (SAS/SATA/NVMe)",
    });
  }

  if (networkIOPS !== undefined) {
    limitPoints.push({
      type: "network",
      iops: networkIOPS,
      factor: "Storage Network",
    });
  }

  // Find minimum (bottleneck)
  const bottleneck = limitPoints.reduce((min, current) =>
    current.iops < min.iops ? current : min
  );

  return {
    bottleneck: bottleneck.type,
    effectiveIOPS: bottleneck.iops,
    limitingFactor: bottleneck.factor,
  };
}

/**
 * Calculate bus IOPS limit based on interface type and drive count.
 *
 * @param busType - Storage bus type (SAS, SATA, NVMe)
 * @param driveCount - Number of drives
 * @param bandwidth - Bus bandwidth in MB/s
 * @param blockSize - Average block size in KB
 * @returns Maximum IOPS sustainable by bus
 *
 * @example
 * // 12Gb SAS with 8 drives, 8KB blocks
 * const busLimit = calculateBusLimit('SAS', 8, 1200, 8)
 * // Returns ~150,000 IOPS (1200 MB/s / 8 KB)
 */
export function calculateBusLimit(
  busType: "SAS" | "SATA" | "NVMe",
  driveCount: number,
  bandwidth: number,
  blockSize: number
): number {
  // Convert MB/s to KB/s, divide by block size
  const busIOPS = (bandwidth * 1024) / blockSize;

  return busIOPS;
}

/**
 * Calculate network IOPS limit for distributed storage.
 *
 * @param networkBandwidth - Network bandwidth in Gbps
 * @param blockSize - Average block size in KB
 * @param protocol - Storage protocol ('iSCSI', 'NFS', 'FC')
 * @returns Maximum IOPS sustainable by network
 *
 * @example
 * // 10GbE network with 8KB blocks over iSCSI
 * const networkLimit = calculateNetworkLimit(10, 8, 'iSCSI')
 * // Returns ~125,000 IOPS (10 Gbps / 8 KB, accounting for iSCSI overhead)
 */
export function calculateNetworkLimit(
  networkBandwidth: number,
  blockSize: number,
  protocol: "iSCSI" | "NFS" | "FC" = "iSCSI"
): number {
  // Protocol overhead factors
  const protocolOverhead: Record<string, number> = {
    iSCSI: 0.85, // ~15% TCP/IP overhead
    NFS: 0.8, // ~20% overhead (TCP/IP + NFS)
    FC: 0.95, // ~5% overhead (minimal Fibre Channel)
  };

  // Convert Gbps to MB/s: Gbps * 125
  const bandwidthMBps = networkBandwidth * 125 * protocolOverhead[protocol];

  // Convert to IOPS based on block size
  const networkIOPS = (bandwidthMBps * 1024) / blockSize;

  return networkIOPS;
}
```
