<objective>
Implement cooperative yielding with abort functionality in Monte Carlo worker and reduce default iteration count for faster user experience.

Purpose: Currently, the Monte Carlo simulation runs 100,000 iterations synchronously, taking 20-30 seconds and blocking abort messages. The abort handler (line 274-276) posts 'ABORTED' but never actually stops the running loop. Users experience unresponsive UI and cannot cancel simulations. Reducing default to 10,000 iterations provides sufficient statistical accuracy (99.87% ± 0.15%) while completing in 1-2 seconds.

Output: Worker that chunks simulations with setTimeout yielding, responds to abort within 1 second, defaults to 10,000 iterations with optional extended analysis mode.
</objective>

   ```typescript
   const CHUNK_SIZE = 1000; // Process 1000 iterations per chunk

   for (let i = 0; i < simulationCount; i += CHUNK_SIZE) {
     // Check abort flag at start of each chunk
     if (shouldAbort) {
       postMessage({ type: "ABORTED" });
       shouldAbort = false; // Reset for next simulation
       return;
     }

     // Process chunk
     const chunkEnd = Math.min(i + CHUNK_SIZE, simulationCount);
     for (let j = i; j < chunkEnd; j++) {
       const result = runSingleSimulation(input);
       // ... accumulate results (same as current logic)

     // Report progress (keep existing logic but with chunkEnd)
     if (chunkEnd % progressInterval === 0 || chunkEnd === simulationCount) {
       postMessage({
         type: "PROGRESS",
         payload: { completed: chunkEnd, total: simulationCount },
       });
     }

     // Yield to event loop (allows abort message to be processed)
     if (chunkEnd < simulationCount) {
       await new Promise((resolve) => setTimeout(resolve, 0));
     }
   }
   ```

   ```typescript
   import { describe, it, expect, vi } from "vitest";

   describe("resilienceWorker abort functionality", () => {
     it("should abort simulation when ABORT message received", async () => {
       // Create worker
       const worker = new Worker(
         new URL("../../src/workers/resilienceWorker.ts", import.meta.url),
         { type: "module" }
       );

       // Track messages
       const messages: any[] = [];
       worker.onmessage = (e) => messages.push(e.data);

       // Start simulation
       worker.postMessage({
         type: "START",
         payload: {
           driveCount: 10,
           raidLevel: "RAID5",
           driveCapacityBytes: 4e12,
           rebuildSpeedMBs: 200,
           ureRate: 14,
           afrPercent: 2,
           simulationCount: 100000, // Large count to ensure abort happens mid-simulation
         },
       });

       // Wait 100ms then abort
       await new Promise((resolve) => setTimeout(resolve, 100));
       worker.postMessage({ type: "ABORT" });

       // Wait for abort confirmation
       await new Promise((resolve) => setTimeout(resolve, 200));

       // Verify ABORTED message received
       const abortedMessage = messages.find((m) => m.type === "ABORTED");
       expect(abortedMessage).toBeDefined();

       // Verify simulation didn't complete (no RESULT message)
       const resultMessage = messages.find((m) => m.type === "RESULT");
       expect(resultMessage).toBeUndefined();

       worker.terminate();
     });

     it("should complete fast with 10K iterations", async () => {
       const worker = new Worker(
         new URL("../../src/workers/resilienceWorker.ts", import.meta.url),
         { type: "module" }
       );

       const startTime = Date.now();

       const result = await new Promise<any>((resolve) => {
         worker.onmessage = (e) => {
           if (e.data.type === "RESULT") resolve(e.data.payload);
         };

         worker.postMessage({
           type: "START",
           payload: {
             driveCount: 10,
             raidLevel: "RAID5",
             driveCapacityBytes: 4e12,
             rebuildSpeedMBs: 200,
             ureRate: 14,
             afrPercent: 2,
             simulationCount: 10000, // Default count
           },
         });
       });

       const duration = Date.now() - startTime;

       // Verify completes within 3 seconds (should be ~1-2s)
       expect(duration).toBeLessThan(3000);

       // Verify result has survival rate
       expect(result.survivalRate).toBeGreaterThan(0);

       worker.terminate();
     });
   });
   ```
