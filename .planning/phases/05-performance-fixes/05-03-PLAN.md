1. Create new component file **src/components/ExportPdfButton.tsx**:
   ```typescript
   /**
    * PDF export button component (lazy-loaded to defer jsPDF bundle).
    * Only loads 407KB of PDF libraries when user first clicks export.
    */

   import { jsPDF } from 'jspdf'
   import autoTable from 'jspdf-autotable'
   import { exportToPdf } from '@/utils/exportPdf'
   import { useConfigStore } from '@/store'
   import drivesData from '@/data/drives.json'
   import type { Drive } from '@/types'
   import type { CalculationResults } from '@/types/results'

   const drives = drivesData as Record<string, Drive>

   interface ExportPdfButtonProps {
     results: CalculationResults
     onClose?: () => void
   }

   export default function ExportPdfButton({ results, onClose }: ExportPdfButtonProps) {
     const config = useConfigStore()
     const selectedDrive = drives[config.driveId] || null

     const handleExport = () => {
       if (!selectedDrive) return

       exportToPdf({
         drive: selectedDrive,
         driveCount: config.driveCount,
         serverCount: config.serverCount,
         topology: config.topology,
         volumetry: results.volumetry,
         performance: results.performance,
         sustainability: results.sustainability,
         unitSystem: config.unitSystem,
       })

       if (onClose) onClose()
     }

     return (
       <button
         onClick={handleExport}
         className="px-4 py-2 bg-primary-600 hover:bg-primary-700 text-white rounded-lg transition-colors"
         disabled={!selectedDrive}
       >
         Download PDF Report
       </button>
     )
   }
   ```

2. Update **src/components/layout/OutputDashboard.tsx**:
   - Find the PDF export button section (search for "exportToPdf")
   - Add lazy import at top of file:
     ```typescript
     import { lazy, Suspense, useState } from 'react'
     const ExportPdfButton = lazy(() => import('../ExportPdfButton'))
     ```
   - Replace direct exportToPdf call with conditional rendering:
     ```typescript
     const [showExport, setShowExport] = useState(false)

     // In JSX where PDF export button currently is:
     {!showExport ? (
       <button
         onClick={() => setShowExport(true)}
         className="px-4 py-2 bg-primary-600 hover:bg-primary-700 text-white rounded-lg"
       >
         Export PDF
       </button>
     ) : (
       <Suspense fallback={<div className="px-4 py-2 text-slate-400">Loading PDF export...</div>}>
         <ExportPdfButton results={results} onClose={() => setShowExport(false)} />
       </Suspense>
     )}
     ```
   - Remove direct import of exportToPdf from @/utils (no longer needed at top level)

**Important:** The pattern is "click Export PDF → show Suspense fallback → load chunk → show ExportPdfButton → click Download → export". This defers 407KB jsPDF bundle until user initiates export.

**Important:** Keep Suspense fallback simple (no heavy components) to ensure fast initial render.
